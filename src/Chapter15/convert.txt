packageChapter15;importjava.io.FileReader;importjava.io.FileWriter;importjava.io.IOException;publicclassExample15_9{publicstaticvoidmain(String[]args){try{//FileReader를사용하여소스파일을읽어옴FileReaderfr=newFileReader(".\\src\\Chapter15\\Example15_9.java");//FileWriter를사용하여변환된내용을저장할파일을생성FileWriterfw=newFileWriter(".\\src\\Chapter15\\convert.txt");intdata=0;//파일을읽어오면서탭,개행,공백,캐리지리턴을제외한문자를FileWriter를사용하여새로운파일에씀while((data=fr.read())!=-1){if(data!='\t'&&data!='\n'&&data!=''&&data!='\r')fw.write(data);}fr.close();//FileReader를닫음fw.close();//FileWriter를닫음}catch(IOExceptione){e.printStackTrace();}}//main}
